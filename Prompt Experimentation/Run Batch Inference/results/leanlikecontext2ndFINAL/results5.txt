\begin{theorem}\label{thm:main}
Let $K_{n}$ be an $r$-edge-colored complete graph with $2\leq r \leq {n \choose 2}$ and $t=f(r)$. Then 
\begin{equation}\label{eq:main}
t_r(K_n)=
\left\{
\begin{array}{ll}

n, & \hbox{otherwise }
\end{array}
\right.
\end{equation}
\end{theorem}
